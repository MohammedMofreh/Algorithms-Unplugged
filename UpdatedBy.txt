columns.Add().RenderedAs(model =>
        {
            var updatedBy = model.UpdatedBy ?? "N/A";
            var updatedDate = model.UpdatedAt.HasValue ? model.UpdatedAt.Value.ToString("dd/MM/yyyy") : "N/A";
            var updatedTime = model.UpdatedAt.HasValue ? model.UpdatedAt.Value.ToString("hh:mm:ss tt") : "N/A";

            return $"{updatedBy}<br/>" +
                   $"{updatedDate}<br/>" +
                   $"{updatedTime}";
        })
        .Encoded(false)
        .Titled("Updated By");
